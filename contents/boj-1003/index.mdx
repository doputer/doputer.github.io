```c
#include <stdio.h>

int mem[100] = {0};
int count[100][2] = {0};

int fibonacci(int n)
{
	if (n == 0)
		return 0;
	else if (n == 1)
		return 1;
	else if (mem[n] != 0)
		return mem[n];
	else
	{
		mem[n] = fibonacci(n - 1) + fibonacci(n - 2);
		count[n][0] = count[n - 1][0] + count[n - 2][0];
		count[n][1] = count[n - 1][1] + count[n - 2][1];

		return mem[n];
	}
}

int main(void)
{
	int T, N;

	count[0][0] = 1;
	count[0][1] = 0;
	count[1][0] = 0;
	count[1][1] = 1;

	scanf("%d", &T);

	for (int i = 0; i < T; i++)
	{
		scanf("%d", &N);
		fibonacci(N);
		printf("%d %d\n", count[N][0], count[N][1]);
	}

	return 0;
}
```
